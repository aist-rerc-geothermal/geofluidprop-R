
file(GLOB FILES_HEADER RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.h)
file(GLOB FILES_SOURCE RELATIVE ${CMAKE_CURRENT_SOURCE_DIR} *.c)
set(TMP_SOURCES ${FILES_HEADER} ${FILES_SOURCE})
if(NOT USE_FREESTEAM)
  list(REMOVE_ITEM TMP_SOURCES "eos-if97-freesteam.h")
  list(REMOVE_ITEM TMP_SOURCES "eos-if97-freesteam.c")
endif()
if(NOT USE_PROST)
  list(REMOVE_ITEM TMP_SOURCES "eos-iaps84-prost.h")
  list(REMOVE_ITEM TMP_SOURCES "eos-iaps84-prost.c")
  list(REMOVE_ITEM TMP_SOURCES "eos-driesner07-prost.h")
  list(REMOVE_ITEM TMP_SOURCES "eos-driesner07-prost.c")
endif()

list(TRANSFORM TMP_SOURCES PREPEND "${CMAKE_CURRENT_SOURCE_DIR}/")
set(SOURCES ${SOURCES} ${TMP_SOURCES} PARENT_SCOPE)
